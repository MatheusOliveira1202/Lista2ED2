-----> C贸digo O(log n ) em R

bin_search <- function (v, t) {
      lo <- 1
      hi <- length(v)
      while (lo <= hi) {
            mid = round((lo + hi) / 2)
            
            if (abs(v[mid] - t) <= 0 ) {
                  return(mid)
                  break
            } else if (v[mid] < t) {
                  lo = mid + 1
            } else {
                  hi = mid - 1
            }
      }
}

<?php
function bubble_Sort($my_array )
{
	do
	{
		$swapped = false;
		for( $i = 0, $c = count( $my_array ) - 1; $i < $c; $i++ )
		{
			if( $my_array[$i] > $my_array[$i + 1] )
			{
				list( $my_array[$i + 1], $my_array[$i] ) =
						array( $my_array[$i], $my_array[$i + 1] );
				$swapped = true;
			}
		}
	}
	while( $swapped );
return $my_array;
}
?>
